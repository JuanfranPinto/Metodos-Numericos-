def gaussian_elimination(A, b):
    n = len(b)
    A = np.hstack((A, b.reshape(-1, 1)))  # Matriz aumentada

    # Eliminaci√≥n hacia adelante (sin redondeo en factor)
    for i in range(n):
        for j in range(i + 1, n):
            factor = A[j, i] / A[i, i]
            A[j, i:] -= factor * A[i, i:]

    # Sustituci√≥n hacia atr√°s con redondeo final
    x = np.zeros(n)
    for i in range(n - 1, -1, -1):
        x[i] = (A[i, -1] - np.dot(A[i, i + 1:n], x[i + 1:])) / A[i, i]

    return np.round(x, 2)  # Redondeo aqu√≠

import numpy as np

def eliminacion_gaussiana(A, b, tol=1e-12, pivoteo=True):
   
    A = np.array(A, dtype=float)
    b = np.array(b, dtype=float).reshape(-1)

    n = A.shape[0]
    if A.shape != (n, n):
        raise ValueError("A debe ser cuadrada (n x n).")
    if b.shape[0] != n:
        raise ValueError("b debe tener longitud n.")

    # Matriz aumentada [A | b]
    
    Ab = np.hstack([A, b.reshape(-1, 1)])

    # Eliminaci√≥n hacia adelante
    
    for i in range(n):
        if pivoteo:
            # Buscar el mayor pivote en columna i desde fila i
            fila_max = i + np.argmax(np.abs(Ab[i:, i]))
            if abs(Ab[fila_max, i]) < tol:
                continue
            
            # Intercambiar filas
            if fila_max != i:
                Ab[[i, fila_max]] = Ab[[fila_max, i]]

        # Si el pivote es ~0, no se puede eliminar en esta columna
        piv = Ab[i, i]
        if abs(piv) < tol:
            continue

        for j in range(i + 1, n):
            factor = Ab[j, i] / piv
            Ab[j, i:] -= factor * Ab[i, i:]

    U = Ab[:, :n]
    c = Ab[:, n]

    
    rango_A = np.linalg.matrix_rank(A, tol=tol)
    rango_Ab = np.linalg.matrix_rank(Ab[:, :], tol=tol)

    if rango_Ab > rango_A:
        return {"estado": "sin_solucion", "x": None, "U": U, "Ab": Ab}
    if rango_A < n:
        return {"estado": "infinitas", "x": None, "U": U, "Ab": Ab}

    # Sustitucion hacia atras
    x = np.zeros(n)
    for i in range(n - 1, -1, -1):
        if abs(U[i, i]) < tol:
            
            return {"estado": "infinitas", "x": None, "U": U, "Ab": Ab}

        suma = np.dot(U[i, i+1:], x[i+1:])
        x[i] = (c[i] - suma) / U[i, i]

    return {"estado": "unica", "x": x, "U": U, "Ab": Ab}

def gaussian_elimination(matrix, vector):
    A = np.array(matrix, dtype=np.float64)
    b = np.array(vector, dtype=np.float64)

    n = len(b)

    # Eliminaci√≥n hacia adelante con pivoteo parcial
    for i in range(n):
        max_row = np.argmax(np.abs(A[i:, i])) + i
        if i != max_row:
            A[[i, max_row]] = A[[max_row, i]]
            b[[i, max_row]] = b[[max_row, i]]

        pivot = A[i, i]
        if np.isclose(pivot, 0):
            raise ValueError("Sistema singular o sin soluci√≥n √∫nica.")

        for j in range(i + 1, n):
            factor = A[j, i] / pivot
            A[j, i:] -= factor * A[i, i:]
            b[j] -= factor * b[i]

    # Sustituci√≥n hacia atr√°s
    x = np.zeros(n)
    for i in range(n - 1, -1, -1):
        x[i] = (b[i] - np.dot(A[i, i + 1:], x[i + 1:])) / A[i, i]

    return np.round(x, 6)  # Ajusta la precisi√≥n final si lo deseas

def analyze_system(alpha):
    # Construye matriz del sistema en funci√≥n de Œ±
    A = np.array([
        [1, -1, alpha],
        [-1, 2, -alpha],
        [alpha, 1, 1]
    ], dtype=np.float64)
    b = np.array([-2, 3, 2], dtype=np.float64)

    # Determinante para saber si tiene soluci√≥n √∫nica
    det = np.linalg.det(A)

    if np.isclose(det, 0):
        # Verificamos los rangos
        augmented = np.column_stack((A, b))
        rank_A = np.linalg.matrix_rank(A)
        rank_aug = np.linalg.matrix_rank(augmented)

        if rank_A == rank_aug:
            return "Infinitas soluciones"
        else:
            return "No tiene soluci√≥n"
    else:
        # Soluci√≥n √∫nica
        x = np.linalg.solve(A, b)
        return np.round(x, 4)

def analizar_sistema(a):
    # Matriz de coeficientes
    A = np.array([
        [1, -1, a],
        [-1, 2, -a],
        [a, 1, 1]
    ], dtype=float)

    # Vector independiente
    b = np.array([-2, 3, 2], dtype=float)

    # Determinante
    det_A = np.linalg.det(A)

    if np.isclose(det_A, 0):
        # Verificamos los rangos
        rA = np.linalg.matrix_rank(A)
        Ab = np.column_stack((A, b))
        rAb = np.linalg.matrix_rank(Ab)

        if rA == rAb:
            return f"Para a = {a:.2f}: Infinitas soluciones"
        else:
            return f"Para a = {a:.2f}: No tiene soluci√≥n"
    else:
        # Soluci√≥n √∫nica
        x = np.linalg.solve(A, b)
        return f"Para a = {a:.2f}: Soluci√≥n √∫nica ‚Üí x = {np.round(x, 3)}"

def calcular_incrementos_individuales(A, x, b):
    A = np.array(A, dtype=float)
    x = np.array(x, dtype=float)
    b = np.array(b, dtype=float)

    incrementos = []
    consumo_actual = A @ x
    disponible = b - consumo_actual

    for j in range(len(x)):
        a_col = A[:, j]
        # Detectamos si alguna entrada de a_col es cero
        with np.errstate(divide='ignore', invalid='ignore'):
            delta = np.where(np.abs(a_col) > 1e-8, disponible / a_col, np.inf)

        # Solo consideramos los l√≠mites v√°lidos
        delta_valido = delta[np.isfinite(delta)]
        if len(delta_valido) > 0:
            incremento_max = np.floor(np.min(delta_valido))
        else:
            incremento_max = 0  # No se puede agregar esa especie

        incrementos.append(int(incremento_max))

    return incrementos

def gauss_jordan(A, b):
    """
    Resuelve un sistema de ecuaciones lineales Ax = b usando el m√©todo Gauss-Jordan.
    """
    A = A.astype(np.float32)
    b = b.astype(np.float32)
    n = len(b)
    
    # Formar la matriz aumentada
    aug_matrix = np.hstack((A, b.reshape(-1, 1)))

    for i in range(n):
        # Hacer el pivote igual a 1 dividiendo toda la fila
        aug_matrix[i] = aug_matrix[i] / aug_matrix[i, i]

        # Hacer ceros en la columna del pivote para las dem√°s filas
        for j in range(n):
            if i != j:
                aug_matrix[j] -= aug_matrix[j, i] * aug_matrix[i]

    # Extraer la soluci√≥n
    x = aug_matrix[:, -1]
    return x
